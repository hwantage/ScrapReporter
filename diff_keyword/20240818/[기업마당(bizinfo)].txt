        var keywords = []; // preKeywords
        var htmlTag = "'";
        // 현재 검색어
        var schKeyword = "".replace("&amp;","&").replace(/amp;/gi,"").replace(/&gt;/gi,">").replace(/&lt;/gi,"<").trim();
        // 이전 검색어
        var preKeywords = ''.replace("&amp;","&").replace(/amp;/gi,"").replace(/&gt;/gi,">").replace(/&lt;/gi,"<");
        function addKwdLst() {
        	if(schKeyword == '' && "N" != 'Y') {
                $("#rescanChk").css("display","none");
            }
        	// 결과 내 재검색
            if("N" == "Y") {
            	 // 검색1,검색2,검색3 ... String으로 넘어온 값을 배열로 파싱
                 if(preKeywords.length > 0) {
                     keywords = preKeywords.split(',');
                 } 
            	 // 재검색 중 이전에 검색한 적 없는 문자열만 배열에 추가
                 if(!keywords.includes(schKeyword)) {
                	keywords.push(schKeyword);
                	$("input[name=preKeywords]").val(keywords);
                 } else {
                     $("input[name=preKeywords]").val("");
                 }
            	
                $("#reSearch").prop("checked",true);
                
            } else if ("N" == "N") {
                keywords.push("");
                $("input[name=preKeywords]").val(keywords);
            } else {
                keywords.push("");
            }
            // 키워드 삭제 버튼 추가
            for(var keyword in keywords) {
            	var kwd = keywords[keyword].replace("&amp;","&").replace(/amp;/gi,"").replace(/&gt;/gi,">").replace(/&lt;/gi,"<");
                htmlTag += kwd
                +"<a href=\"#\" onclick=\"javascript:removeKwdLst('"+kwd+"');\">" 
                  +"<img class='keyword_del_btn' style='position: relative; bottom: 1px; left: 1px;' alt='"+kwd+" 삭제' src=\"/images/bizinfo/common/btn_delete_02.gif\"/>"
                   +"</a>,";
            }
            // 마지막 콤마 제거 정규식 (str1,str2,str3, -> str1,str2,str3)
            htmlTag = htmlTag.replace(/,\s*$/, "");
            htmlTag += "'";
            $('#reSearchKeywords').html(htmlTag);
            
            // 마지막 키워드 삭제 버튼 제거
            /* $('.keyword_del_btn').last().remove(); */
        }
        
        
        // 결과 내 재검색 검색어 삭제
        function removeKwdLst(keyword){
        	if(keywords.length <= 1) {
                alert('검색어는 한 개 이상 존재해야 합니다.');
                return;
            }
        	// 중복 데이터 배열에서 제거
        	keywords = keywords.filter((kwd) => kwd != keyword);
           	$("input[name=preKeywords]").val(keywords);
           	$("input[name=keyword]").val(keywords[keywords.length - 1]);
        	articleSearch();
        }
