        function getTabInTabs(menuCode) {
            let tab = tabList.find(tab => tab.menuCode === menuCode);
            if (tab.subTabList.length > 0) {
                let tabInTabHtml = '';
                let seeAllHtml = '';
                tab.subTabList.forEach((subTab, index) => {

                    if (isSelectedExist) {
                        tabInTabHtml += '<div class="swiper-slide ' + (subTab.selected ? 'active' : '') + '" id="tab_list_0' + (index + 1) + '" aria-selected="' + (subTab.selected ? 'true' : 'false') + '" aria-controls="panel_list_0' + (index + 1) + '">';
                    } else {
                        if (index === 0) {
                            tabInTabHtml += '<div class="swiper-slide active" id="tab_list_0' + (index + 1) + '" aria-selected="true" aria-controls="panel_list_0' + (index + 1) + '">';
                        } else {
                            tabInTabHtml += '<div class="swiper-slide" id="tab_list_0' + (index + 1) + '" aria-selected="false" aria-controls="panel_list_0' + (index + 1) + '">';
                        }
                    }
                    tabInTabHtml += '<button type="button" class="btn-tab" onclick="goToTabMenu(\'' + subTab.userMenuUrl + '\')" data-menu-cd="' + subTab.menuCode + '">' + subTab.menuNm + '' + (subTab.selected ? '<i class="sr-only created"> 선택됨</i>' : '') + '</button>';
                    tabInTabHtml += '</div>';
                    seeAllHtml += '<li>';
                    seeAllHtml += '<div class="opt" data-id="tab_list_0' + (index + 1) + '" aria-selected="' + (subTab.selected ? 'true' : 'false') + '" aria-controls="panel_list_0' + (index + 1) + '">';
                    seeAllHtml += '<button type="button" class="btn-tab" onClick="goToTabMenu(\'' + subTab.userMenuUrl + '\')" data-menu-cd="' + subTab.menuCode + '">' + subTab.menuNm + '' + (subTab.selected ? '<i class="sr-only created"> 선택됨</i>' : '') + '</button>';
                    seeAllHtml += '</div>';
                });
                if (tabInTabHtml !== '') {
                    $('#tab-in-tab').css({'visibility': 'initial'});

                    $('#tab-in-tab-swiper').html(tabInTabHtml);
                    $('#tab-in-tab-ul').html(seeAllHtml);
                } else {
                    $('#tab-in-tab').css({'visibility': 'hidden'});
                }
            }
        }

        function goToTabMenu(userMenuUrl) {
            let isTabInTab = false;

            const target = $(event.target);

            if (!target.closest("#tab").length) {
                isTabInTab = true;
            }

            const selectedIndex = target.closest("div").index();

            const visibleFirstIndex = target.closest("div").closest(".swiper-wrapper").find(".swiper-slide-active").closest("div").index();

            const visibleOrder = Math.abs(visibleFirstIndex - selectedIndex);

            if (isTabInTab) {
                setTabInTabSlideIndex(visibleOrder);
            } else {
                setTabSlideIndex(visibleOrder);
            }

            location.href = userMenuUrl;
        }

    function goToSearch(query) {
        let form = document.createElement("form");
        form.setAttribute("method", "POST");
        form.setAttribute("action", "https://search.mss.go.kr/mssSearch/search.do");
        form.setAttribute("target", "_blank");

        let hiddenField = document.createElement("input");
        hiddenField.setAttribute("type", "hidden");
        hiddenField.setAttribute("name", "query");
        hiddenField.setAttribute("value", query);

        form.appendChild(hiddenField);

        document.body.appendChild(form);
        form.submit();
    }

function setAWeekAgo(){
	var today = new Date();
	var returnDate = new Date();
	returnDate.setFullYear(today.getFullYear(), today.getMonth(), today.getDate()-7);
	var y = returnDate.getFullYear();
	var m = returnDate.getMonth() + 1;
	var d = returnDate.getDate();
	if(m < 10) { m = "0" + m; }
	if(d < 10) { d = "0" + d; }
	var resultDate = y + m + d;
	return resultDate;
}
