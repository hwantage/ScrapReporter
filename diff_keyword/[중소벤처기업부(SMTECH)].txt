	function fn_saveLginId() {
		var pwd = $("#chngLginPw").val();
		var rePwd = $("#reChngLginPw").val();
		
		if (!$("#chngLginId").val()) {
			alert("아이디를 입력해주세요.");
			return;
		} else if ($("#dupLginChkYn").val() != "Y") {
			alert("사용할 수 없는 아이디입니다.");
			$("#chngLginId").focus();
			return
		} else if (!$("#chngLginPw").val()) {
			alert("비밀번호를 입력해주세요.");
			$("#chngLginPw").focus();
			return;
		} else if (!$("#reChngLginPw").val()) {
			alert("비밀번호 확인을 입력해주세요.");
			return;
		} else if (pwd != rePwd) {
			alert("비밀번호와 비밀번호 확인 정보가 서로 다릅니다.");
			$("#chngLginPw").focus();
			return;
		}
		
		const isPwRegObj = {
				// 영문대소문자, 숫자, 특수문자 조합
				fullReg: /^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*#=+/;:{}\(\)|\[\]^`,.~_-])[A-Za-z\d@$!%*#=+/;:{}\(\)|\[\]^`,.~_-]{10,20}$/,
				fullRegMsg: "비밀번호는 10~20자리, 영문 대소문자, 숫자, 특수문자를 결합하여 사용해야 합니다.\n\n& ' \" < > ? 문자는 사용불가"
		};
		
		if (!isPwRegObj.fullReg.test(pwd) || !isPwRegObj.fullReg.test(rePwd)) {
			alert(isPwRegObj.fullRegMsg);
			$("#chngLginPw").focus();
			return;
		}
						
		
		
		var key = "0101110101011111";
		var iv = "0101110101011111";
		
		var keyUtf = CryptoJS.enc.Utf8.parse(key);
		var ivUtf = CryptoJS.enc.Utf8.parse(iv);
		
		var idEnc = CryptoJS.AES.encrypt($("#chngLginId").val(), keyUtf, {iv: ivUtf});
		var pwEnc = CryptoJS.AES.encrypt($("#chngLginPw").val(), keyUtf, {iv: ivUtf});
		
		// 저장로직
		$.ajax({
			type: "POST",
			url: "/front/log/changeLginId.do",
			data: "LGIN_ID=" + encodeURIComponent(idEnc)
					+ "&PIN_NO=" + encodeURIComponent(pwEnc),
			dataType: "json",
			success: function(data){
				if (data && data.SUCCESS_YN == "Y") {
					alert("계정정보 저장이 완료되었습니다.");
					
					
					
					location.href = "/front/mpa/ip/IndyPimsRead.do";
					
				} else {
					alert("계정정보 저장에 실패하였습니다.");
				}
			},
			error: function(jqXHR,textStatus,errorThrown){
				alert("계정정보 저장에 실패하였습니다.");
			}
		});
		
	}
