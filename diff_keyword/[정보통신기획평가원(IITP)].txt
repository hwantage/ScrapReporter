<script>
	$(document).ready(function () {
		var $html = $('html');
		var savedSize = localStorage.getItem('zoomSize');
		var savedMode = localStorage.getItem('viewMode');

		if (savedSize) {
			applyZoom(savedSize);
			$('.radio_inp[value="' + savedSize + '"]').prop('checked', true);
		}

		if (savedMode) {
			applyViewMode(savedMode);
			$('.radio_inp[value="' + savedMode + '"]').prop('checked', true);
		} else {
			applyViewMode('default');
		}

		$('.screen_pop_area > .inner_box > .box.left .radio_list_area .list li .radio_area label .radio_inp').on('click', function () {
			var $inp = $(this).val();
			applyZoom($inp);
			localStorage.setItem('zoomSize', $inp);
		});

		function applyZoom(size) {
			var zoomLevel = {
				'small': '0.9',
				'medium': '1',
				'large': '1.1',
				'xlarge': '1.3',
				'xxlarge': '1.5'
			};

			if (zoomLevel[size]) {
				$html.css('zoom', zoomLevel[size]);
			}
		}

		$('.radio_inp[name="view_mode"]').on('change', function () {
			var selectedMode = $(this).val();
			applyViewMode(selectedMode);
			localStorage.setItem('viewMode', selectedMode);
		});

		function applyViewMode(mode) {
			if (mode === 'light') {
				$html.removeClass('dark-mode').addClass('light-mode');
			} else if (mode === 'dark') {
				$html.removeClass('light-mode').addClass('dark-mode');
			} else {
				if (window.matchMedia('(prefers-color-scheme: dark)').matches) {
					$html.removeClass('light-mode').addClass('dark-mode');
				} else {
					$html.removeClass('dark-mode').addClass('light-mode');
				}
			}
		}

		$('header .mode_btn').on('click', function (){
			$('.screen_pop_wrap').addClass('active');
			$('.screen_pop_wrap .box.left .radio_list_area li:first-child input').focus();
			$('li.lang_li .list_area').slideUp(300);
			$('.lang_btn').removeClass('on');
		});

		$('.screen_pop_wrap .close').on('click', function (){
			$(this).closest('.screen_pop_wrap').removeClass('active');

			$('.mode_btn').focus();
		});

		windowW = window.innerWidth;
		if(windowW>1200){
			$('header .mode_btn').on('click', function (){
				$('body').attr('style', 'overflow: hidden;');

				$('html').css('padding-right', '15px');
				$('header').css('margin-right', '-15px');
				$('header .h_unri_area').css('padding-right', '15px');
				$('header .header_area .header_inner').css('padding', '17px 15px 24px 0');
				$('header .header_area #lnb').css('padding-right', '15px');

				$('footer').css('margin-right', '-15px');
				$('footer .ft_site_area').css('padding', '34px 15px 34px 0');
				$('footer .ft_link_area').css('padding-right', '15px');
				$('footer .footer_area').css('padding', '45px 15px 25px 0');
				$('footer .f_unri_area').css('padding-right', '15px');
			});

			$('.screen_pop_wrap .close').on('click', function (){
				$('html').css('padding-right', '0');
				$('header').css('margin-right', '0');
				$('header .h_unri_area').css('padding-right', '0');
				$('header .header_area .header_inner').css('padding', '17px 0 24px');
				$('header .header_area #lnb').css('padding-right', '0');
				$('body').attr('style', 'overflow: visible;');

				$('footer').css('margin-right', '0');
				$('footer .ft_site_area').css('padding', '34px 0');
				$('footer .ft_link_area').css('padding-right', '0');
				$('footer .footer_area').css('padding', '45px 0 25px');
				$('footer .f_unri_area').css('padding-right', '0');

			});
		}
		else{
		}

		$('header .sch_btn').on('click', function (){
			$('.header_sch_wrap').addClass('active');
			$('body').attr('style', 'overflow: hidden;');
			$('.help_btn').focus();
		});

		$('.header_sch_wrap .h_sch_close').on('click', function (){
			$('.header_sch_wrap').removeClass('active');
			$('body').attr('style', 'overflow: visible;');
			$('header .sch_btn').focus();
		});


		$('.all_menu_wrap_m').hide();
		windowW = window.innerWidth;
		if(windowW>1023){
