	//타이머
	function fn_timer() {

		sec = sec - 1;
		if(sec == -1) {
			min = min-1;
			sec = 59;
		}
		cTime = window.setTimeout('fn_timer()', 1000);
		if(min > 0) {
			//if(min == 1 && sec == 50) {
			if(min == 10 && sec == 0){
				sessionIngPopup();
			}

			if(sec > 9) {
				$('#inactive_time').html("<b>" + min + ":" + sec + "</b>");

				
				if($("#sessionAlarmPop").css("visibility") == "visible") {
					$("#inactive_pop_time").html("<b><font size='4' color='red'>만료시간이 " + min + "분 " + sec + "초 남았습니다.</font></b>");
				}

			} else {
				$('#inactive_time').html("<b>" + min + ":0" + sec + "</b>");
				if($("#sessionAlarmPop").css("visibility") == "visible") {
					$("#inactive_pop_time").html("<b><font size='4' color='red'>만료시간이 " + min + "분 0" + sec + "초 남았습니다.</font></b>");
				}
			}
		} else {
			if(sec >= 1) {
				$('#inactive_time').html("<b>0:" + sec + "</b>");
				if($("#sessionAlarmPop").css("visibility") == "visible") {
					$("#inactive_pop_time").html("<b><font size='4' color='red'>만료시간이 " + "0분 " + sec + "초 남았습니다.</font></b>");
				}
			} else {
				$('#inactive_time').html("<b>0:00</b>");
				if($("#sessionAlarmPop").css("visibility") == "visible") {
					$("#inactive_pop_time").html("<b><font size='4' color='red'>시간이 만료되었습니다.</font></b>");
				}
				alert("로그인 시간이 만료되어 로그아웃됩니다.");
				clearTimeout(cTime);

				location = "/front/log/logoutProc.do";
				

			}
		}
	}
