function fn_date_set(gubun){
	var startDt = "";
	var endDt = "";
	var today = new Date();
	var y = today.getFullYear();
	var m = today.getMonth() + 1;
	var d = today.getDate();
	if(m < 10) { m = "0" + m; }
	if(d < 10) { d = "0" + d; }
	var endDt = y + m + d;
	if(gubun == 'w'){
		startDt = setAWeekAgo();
	}else if(gubun == 'm1'){
		startDt = setAMonthAgo();
	}else if(gubun == 'm3'){
		startDt = setAMonth3Ago();
	}else{
		startDt = setAMonthAgo();
	}
	
	$("#regSdt").val(startDt);
	$("#regEdt").val(endDt);
}


//일주일전 날짜 구하기
function setAWeekAgo(){
	var today = new Date();
	var returnDate = new Date();
	returnDate.setFullYear(today.getFullYear(), today.getMonth(), today.getDate()-7);
	var y = returnDate.getFullYear();
	var m = returnDate.getMonth() + 1;
	var d = returnDate.getDate();
	if(m < 10) { m = "0" + m; }
	if(d < 10) { d = "0" + d; }
	var resultDate = y + m + d;
	return resultDate;
}
 
//한달전 날짜 구해오기

	var today = new Date();

	returnDate.setFullYear(today.getFullYear(), today.getMonth()-1, today.getDate());
	var y = returnDate.getFullYear();
	var m = returnDate.getMonth() + 1;
	var d = returnDate.getDate();
	if(m < 10) { m = "0" + m; }
	if(d < 10) { d = "0" + d; }
	var resultDate = y + m + d;
	return resultDate;
}
