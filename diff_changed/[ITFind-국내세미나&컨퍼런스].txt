function createMenu(menuList, depth, type) {
	var ul = $('<ul></ul>');
	
	if (type == "TOP") {
		if (depth == 1) {
			ul.addClass("itf-rd-navbar-nav");
		} else if (depth == 2) {
			ul.addClass("rd-navbar-dropdown");
			ul.addClass("twodepth-list");
		} else if (depth == 3) {
			ul.addClass("rd-navbar-dropdown");
			ul.addClass("threedepth-list");
		}
	}
	
	menuList.forEach(function(menu){
		var li = $('<li><a href="' + menu.menuUrl + '">' + menu.menuName + '</a></li>');
		
		if (menu.menuUrl.includes("http")) {
			li.find("a").attr("target", "_blank");
		}
		
		if (menu.childMenu != null) {
			li.append(createMenu(menu.childMenu, depth + 1, type));
		}
		
		ul.append(li);
		
		if (/[a-zA-Z]/.test(menu.menuName)) {
			if (15 <= menu.menuName.length) {
				ul.addClass("thm02");
			}
		} else if (10 <= menu.menuName.length) {
			ul.addClass("thm02");
		}
	});
	
	return ul;
}
