<script>
    $(function () {
        $('._oss_tab').each(function () {
            var $container = $(this);
            var filter = function () {
                return $container[0] === $(this).parents('._oss_tab')[0];
            };
            var choose = function (to) {
                $('._oss_tab-item', $container).filter(filter).eq(to).show();
                $('._oss_tab-item', $container).filter(filter).not(':eq('+to+')').hide();
            };

            $('._oss_tab-indicators', this).filter(filter).each(function () {
                var indicators = this;
                $('[data-tab-to]', this).click(function (e) {
                    e.preventDefault();

                    var to = $(this).data('tabTo');
                    choose(to);

                    $(this).addClass('on');
                    $('[data-tab-to]', indicators).not(this).removeClass('on');
                });
            });

            $('._oss_tab-indicators', this).filter(filter).find('[data-tab-to="0"]').triggerHandler('click');

        });
    });
    
</script>
