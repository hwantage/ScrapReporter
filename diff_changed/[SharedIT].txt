document.addEventListener('DOMContentLoaded', function() {
  // 쿠키 설정하기
  function setCookie(name, value, days) {
    let expires = "";
    if (days) {
      let date = new Date();
      date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
      expires = "; expires=" + date.toUTCString();
    }
    document.cookie = name + "=" + (value || "") + expires + "; path=/; Secure; SameSite=Lax";
  }

  // 쿠키에서 데이터를 가져오기
  function getCookie(name) {
    let nameEQ = name + "=";
    let ca = document.cookie.split(';');
    for (let i = 0; i < ca.length; i++) {
      let c = ca[i];
      while (c.charAt(0) == ' ') c = c.substring(1, c.length);
      if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
    }
    return null;
  }

  // 쿠키를 초기화하기
  function resetCookieIfNewDay() {
    const today = new Date().toISOString().split('T')[0];
    const cookieName = `viewed_data`;
    const viewedData = getCookie(cookieName);
    if (!viewedData || JSON.parse(viewedData).date !== today) {
      setCookie(cookieName, JSON.stringify({ date: today, data: {} }), 1);
    }
  }

  // 데이터를 확인했을 때 쿠키 업데이트
  function markAsViewed(dataType, dataIds) {
    const cookieName = `viewed_data`;
    let viewedData = getCookie(cookieName);
    let viewedDataObj = viewedData ? JSON.parse(viewedData) : { data: {} };

    if (!viewedDataObj.data[dataType]) {
      viewedDataObj.data[dataType] = [];
    }

    dataIds.forEach(dataId => {
      if (!viewedDataObj.data[dataType].includes(dataId.toString())) {
        viewedDataObj.data[dataType].push(dataId.toString());
      }
    });

    setCookie(cookieName, JSON.stringify(viewedDataObj), 1); // 1일 동안 유효한 쿠키
  }

  // 새로운 데이터를 체크하여 배지 표시
  function checkNewData() {
    const cookieName = `viewed_data`;
    const viewedData = getCookie(cookieName);
    const viewedDataObj = viewedData ? JSON.parse(viewedData) : { data: {} };
    const dataTypes = ['qnaboards', 'freeboards', 'posts', 'data_centers', 'promotions', 'seminars', 'solutions'];

    dataTypes.forEach(dataType => {
      let newDataExists = window.newData[dataType].some(data => !viewedDataObj.data[dataType] || !viewedDataObj.data[dataType].includes(data.id.toString()));

      if (newDataExists) {
        document.querySelector(`#${dataType}-badge`).style.display = 'inline';
      } else {
        document.querySelector(`#${dataType}-badge`).style.display = 'none';
      }
    });
  }

  window.newData = {"qnaboards":[],"freeboards":[],"posts":[],"data_centers":[],"promotions":[],"seminars":[],"solutions":[]};

  resetCookieIfNewDay();
  checkNewData();
